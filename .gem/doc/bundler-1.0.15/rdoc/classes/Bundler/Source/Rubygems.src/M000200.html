      def initialize(options = {})
        @options = options
        @remotes = (options["remotes"] || []).map { |r| normalize_uri(r) }
        @allow_remote = false
        @allow_cached = false

        # Hardcode the paths for now
        @caches = [ Bundler.app_cache ] + Bundler.rubygems.gem_path.map do |x|
          File.expand_path("#{x}/cache")
        end

        @spec_fetch_map = {}
      end
