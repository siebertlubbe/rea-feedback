    def source(source, options = {})
      case source
      when :gemcutter, :rubygems, :rubyforge then
        rubygems_source "http://rubygems.org"
        return
      when String
        rubygems_source source
        return
      end

      @source = source
      options[:prepend] ? @sources.unshift(@source) : @sources << @source

      yield if block_given?
      @source
    ensure
      @source = nil
    end
