      def set_color(string, color, bold=false)
        color = self.class.const_get(color.to_s.upcase) if color.is_a?(Symbol)
        bold, end_bold = bold ? [BOLD, END_BOLD] : ['', '']
        "#{bold}#{color}#{string}#{CLEAR}#{end_bold}"
      end
